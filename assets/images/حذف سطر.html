from docx import Document
import shutil

python
from docx.oxml import OxmlElement
from docx.oxml.ns import qn

def remove_bottom_border(paragraph):
    p = paragraph._p
    pPr = p.get_or_add_pPr()
    pBdr = pPr.find(qn('w:pBdr'))
    if pBdr is not None:
        for border_name in ['w:bottom']:
            border = pBdr.find(qn(border_name))
            if border is not None:
                pBdr.remove(border)

# Reopen the document
doc = Document(output_path)

# Look for the paragraph that contains "مدير الفرع" and remove bottom border (if any)
for para in doc.paragraphs:
    if "مدير الفرع" in para.text:
        remove_bottom_border(para)

# Save changes
doc.save(output_path)
